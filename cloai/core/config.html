108def initialize_logger() -> logging.Logger:
109    """Initializes the logger for the API."""
110    settings = get_settings()
111    logger = logging.getLogger(settings.LOGGER_NAME)
112    if settings.LOGGER_VERBOSITY is not None:
113        logger.setLevel(settings.LOGGER_VERBOSITY)
114
115    formatter = logging.Formatter(
116        "%(asctime)s - %(name)s - %(levelname)s - %(filename)s:%(lineno)s - %(funcName)s - %(message)s",  # noqa: E501
117    )
118
119    handler = logging.StreamHandler()
120    handler.setFormatter(formatter)
121    logger.addHandler(handler)
122    return logger

125def get_logger() -> logging.Logger:
126    """Gets the logger for the API.
127
128    Returns:
129        The logger for the API.
130    """
131    settings = get_settings()
132    logger = logging.getLogger(settings.LOGGER_NAME)
133    if logger.hasHandlers():
134        return logger
135    return initialize_logger()
